## Types

```tsx
interface AccountTransactionActivity extends ActivityItem {
  /* Transaction type */
  type: AccountTransactionType;
  /* Transaction status */
  status: 'complete';
  /* Transaction icon */
  iconSymbol?: string;
  /* Account transaction notional symbol*/
  notionalSymbol?: string;
  /* Notional rate from base currency at time of transaction */
  notionalRate?: number;
  /* Notional places for the notional amount */
  notionalPlaces?: number;
  /* Account transaction reference type */
  referenceType: ReferenceType;
  /* Account transaction transaction type */
  transactionType: TransactionType;
  /* Is credit entry */
  isCR: boolean;
}

type AccountTransactionType =
  | 'deposit'
  | 'withdraw'
  | 'buy'
  | 'sell'
  | 'send'
  | 'receive'
  | 'deposit-fee'
  | 'withdraw-fee'
  | 'buy-fee'
  | 'sell-fee'
  | 'unknown';

type TransactionType =
  | 'Fee'
  | 'Trade'
  | 'Other'
  | 'Reverse'
  | 'Hold'
  | 'Rebate'
  | 'MarginAcquisition'
  | 'MarginRelinquish';

type ReferenceType =
  | 'Trade'
  | 'Deposit'
  | 'Withdraw'
  | 'Transfer'
  | 'OrderHold'
  | 'WithdrawHold'
  | 'DepositHold'
  | 'MarginHold'
  | 'ManualHold'
  | 'ManualEntry'
  | 'MarginAcquisition'
  | 'MarginRelinquish'
  | 'MarginQuoteHold';
```
